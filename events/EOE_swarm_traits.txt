namespace = eoeswarm
country_event = {
	id = eoeswarm.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		owner = {
			has_ethic = ethic_gestalt_consciousness
			}		
	}
	immediate = {
		every_owned_pop = {
			if = {
				limit = { 
				has_trait = trait_ice_type_mega_evolution
					} 
				
				modify_species = {
					species = this
					add_trait = trait_ice_type_mega_evolution_swarm
					remove_trait = trait_ice_type_mega_evolution
				}
			}
			}
		
			
		}
				
}
