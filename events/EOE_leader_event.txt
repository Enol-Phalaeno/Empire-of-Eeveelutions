fleet_event = {
	id = leadereevent.6
	title = "leadereevent.6.name"
	desc = "leadereevent.6.desc"
	picture = GFX_evt_exploding_ship
	show_sound = event_scanner
	location = from
	is_triggered_only = yes
	fire_only_once = yes
	trigger = {
		OR = {
			is_ship_class = shipclass_science_ship
			is_ship_class = shipclass_military
			is_ship_class = shipclass_constructor
		}
		root.owner = {
			any_owned_leader = {
				has_leader_flag = EOE_leader_elena
			}
			has_country_flag = completed_old_creators
			has_country_flag = leadereevent5_has_happened
			NOR = {
				has_country_flag = leadereevent6_has_happened
				has_civic = civic_slaver_guilds
			}
		}
		solar_system = {
			any_system_planet = {
				is_planet_class = pc_asteroid
			}
		}
	}
	immediate = {
		owner = {
			set_timed_country_flag = {
				flag = EOE_hezi_cooldown
				days = 1800
			}
		}
		solar_system = {
			random_system_planet = {
				limit = {
					is_planet_class = pc_asteroid
				}
				save_event_target_as = distress_singal_source
			}
		}
	}
	after = {
		#name = leadereevent.6.a
		if = {
			limit = {
				is_ship_class = shipclass_science_ship
			}
			root.owner = {
				enable_special_project = {
					name = "EOE_distress_signal_science"
					location = event_target:distress_singal_source
					owner = this
				}
			}
		}
		if = {
			limit = {
				is_ship_class = shipclass_military
			}
			root.owner = {
				enable_special_project = {
					name = "EOE_distress_signal_military"
					location = event_target:distress_singal_source
					owner = this
				}
			}
		}
		if = {
			limit = {
				is_ship_class = shipclass_constructor
			}
			root.owner = {
				enable_special_project = {
					name = "EOE_distress_signal_constructor"
					location = event_target:distress_singal_source
					owner = this
				}
			}
		}
		owner = {
			set_country_flag = leadereevent6_has_happened
		}
	}
}
