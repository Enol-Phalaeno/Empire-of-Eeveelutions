#抓鬼陆军，秋荷一起上的时候抓到的更强
ship_event = {
	
	id = EOE_ghost.3
	is_triggered_only = yes
	title = EOE_ghost.3.name
	desc = {
		trigger = {
			is_hezi_in_shroud = yes
			is_okra_in_shroud = yes
		}
		text = EOE_ghost.3.desc.a
	}
	desc = {
		trigger = {
			NOT = {
				is_hezi_in_shroud = yes
				is_okra_in_shroud = yes
			}
		}
		text = EOE_ghost.3.desc.b
	}
	picture = GFX_evt_EOE_the_ghost
	show_sound = event_mystic_reveal_02
	option = {
		name = EOE_ghost.3.a
		custom_tooltip = EOE_ghost.3.a.tooltip
		hidden_effect = {
			if = {
				limit = {
					is_hezi_in_shroud = yes
					is_okra_in_shroud = yes
				}
				owner = {
					random_owned_planet = {
						limit = { has_planet_flag = EOE_ghost_planet_1 }
						create_army = {
							name = "NAME_Avatar"
							owner = root.owner
							type = "EOE_shroud_army_2"
						}
					}
				}
				
			}
			else = {
				owner = {
					random_owned_planet = {
						create_army = {
							name = "NAME_Avatar"
							owner = root.owner
							type = "EOE_shroud_army"
						}
					}
				}
			}
		}
	}
}

ship_event = {
	id = EOE_ghost.5
	is_triggered_only = yes
	title = EOE_ghost.5.name
	desc = EOE_ghost.5.desc
	picture = GFX_evt_EOE_dangerous_shrouded_planet
	show_sound = event_mystic_reveal_light
	option = {
		#公布发现
		name = EOE_ghost.5.a
		owner = {
			add_modifier = {
				modifier = EOE_fear_of_god
				days = 3600
			}
			if = {
				limit = {
					owner = {
						OR = {
							has_ethic = ethic_spiritualist
							has_ethic = ethic_fanatic_spiritualist
						}
					}
				}
				owner = {
					add_resource = {
						unity = -10000
					}
				}
			}
			else = {
				owner = {
					add_resource = {
						unity = @tier5unityreward
					}
				}
			}
		}
	}
	option = {
		#封锁消息
		name = EOE_ghost.5.b
	}
}
